#include<iostream>
using namespace std;


int main()
{
  int N=4;
    
    for(int i=0;i<N;i++)
    {  
        for(int j=0;j<=(N-i);j++)
        {
            cout<<" ";
        }
        for(int j=0;j<=i*2;j++)
        {
            cout<<"*";
        }
        cout<<endl;
    }
return 0;
    
}
